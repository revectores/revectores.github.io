\begin{algorithmic}[1]
\Function{Doolittle\_decomposition}{$A$, $n$}
\For{$k \gets 1$ to $n$}
    \For{$j \gets k$ to $n$}
        \State $u_{kj} \gets a_{kj} - \displaystyle\sum_{r=1}^{k-1}l_{kr}u_{rj}$
    \EndFor
    \For{$i \gets k+1$ to $n$}
        \State $l_{ik} \gets \displaystyle\frac{1}{u_{kk}}\left( a_{ik} - \sum_{r=1}^{k-1}l_{ir}u_{rk} \right)$
    \EndFor
\EndFor

\State \Return $L, U$

\EndFunction
\end{algorithmic}
